updateStat

Javascript

function updateStat(data, kind) {
  var arr = data[kind];
  var correct = arr[0], guesses = arr[1];
  var percentage = 0;
  if (guesses > 0) {
    percentage =
      parseInt(correct / guesses * 100);
  }
  var text = correct + '/' + guesses
      + ' = ' + percentage + '%'
  $('#' + kind + '-count').text(text);
}
